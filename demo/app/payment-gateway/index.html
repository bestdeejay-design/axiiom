Code Example

5-Minute Integration

// Инициализация платёжного шлюза const gateway = new PaymentGateway({ apiKey: 'your_api_key', merchantId: 'merchant_123', environment: 'sandbox' // sandbox | production }); // Создание платежа const payment = await gateway.createPayment({ amount: 1500.00, currency: 'RUB', description: 'Заказ №4281', split: [ { merchant: 'submerchant_1', amount: 1200.00 }, { merchant: 'submerchant_2', amount: 300.00 } ], escrow: { holdPeriod: 3 } // дней }); // Перенаправление на платёжную страницу window.location.href = payment.paymentUrl;
