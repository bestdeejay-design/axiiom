5-Minute Integration

// Получение метрик за период const metrics = await analytics.getMetrics({ period: 'last_30_days', metrics: ['dau', 'mau', 'conversion', 'ltv'], groupBy: 'day' }); // Построение воронки const funnel = await analytics.buildFunnel({ steps: ['visit', 'register', 'deposit', 'purchase'], startDate: '2024-01-01', endDate: '2024-06-30' }); // Cohort Analysis const cohorts = await analytics.cohortAnalysis({ period: 'monthly', metric: 'retention' });
