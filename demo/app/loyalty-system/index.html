Code Example

5-Minute Integration

// Создание правила лояльности const rule = await loyalty.createRule({ name: 'Buy 3 Get 1 Free', trigger: 'purchase', condition: { category: 'coffee', minItems: 3 }, reward: { type: 'free_product', productId: 'coffee_any' } }); // Сегментация пользователей const segment = await loyalty.createSegment({ name: 'VIP осень 2024', filters: [ { spent: { gt: 50000 } }, { visits: { gt: 10 } } ] }); // Запуск A/B-теста const ab = await loyalty.startABTest({ name: 'Cashback 5% vs 10%', segments: ['A', 'B'], duration: 14 });
